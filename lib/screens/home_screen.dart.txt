import 'package:flutter/material.dart';
import 'package:provider/provider.dart';
import '../models/models.dart';
import '../providers/notebook_provider.dart';

class HomeScreen extends StatelessWidget {
  const HomeScreen({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Theme.of(context).colorScheme.background,
      appBar: AppBar(
        title: Row(
          children: [
            Text(
              'Awalys',
              style: TextStyle(
                fontSize: 28,
                fontWeight: FontWeight.bold,
                color: Theme.of(context).colorScheme.primary,
              ),
            ),
            const SizedBox(width: 8),
            const Text('🧠', style: TextStyle(fontSize: 24)),
          ],
        ),
        elevation: 0,
        backgroundColor: Colors.transparent,
      ),
      body: Consumer<NotebookProvider>(
        builder: (context, provider, child) {
          if (provider.notebooks.isEmpty) {
            return Center(
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Icon(Icons.book_outlined, size: 120, color: Colors.grey[300]),
                  const SizedBox(height: 24),
                  Text(
                    'Nenhum caderno ainda',
                    style: TextStyle(fontSize: 24, fontWeight: FontWeight.w500, color: Colors.grey[600]),
                  ),
                  const SizedBox(height: 12),
                  Text(
                    'Toque no + para criar seu primeiro caderno',
                    style: TextStyle(fontSize: 16, color: Colors.grey[500]),
                  ),
                ],
              ),
            );
          }

          return GridView.builder(
            padding: const EdgeInsets.all(16),
            gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
              crossAxisCount: 2,
              childAspectRatio: 0.85,
              crossAxisSpacing: 16,
              mainAxisSpacing: 16,
            ),
            itemCount: provider.notebooks.length,
            itemBuilder: (context, index) {
              final notebook = provider.notebooks[index];
              return GestureDetector(
                onTap: () {
                  ScaffoldMessenger.of(context).showSnackBar(
                    SnackBar(content: Text('Caderno: ${notebook.title}')),
                  );
                },
                child: Container(
                  decoration: BoxDecoration(
                    color: notebook.color.withOpacity(0.1),
                    borderRadius: BorderRadius.circular(16),
                    border: Border.all(color: notebook.color.withOpacity(0.3), width: 2),
                  ),
                  child: Column(
                    mainAxisAlignment: MainAxisAlignment.center,
                    children: [
                      Container(
                        width: 80,
                        height: 80,
                        decoration: BoxDecoration(
                          color: notebook.color.withOpacity(0.2),
                          borderRadius: BorderRadius.circular(20),
                        ),
                        child: Center(
                          child: Text(notebook.emoji, style: const TextStyle(fontSize: 48)),
                        ),
                      ),
                      const SizedBox(height: 16),
                      Padding(
                        padding: const EdgeInsets.symmetric(horizontal: 12),
                        child: Text(
                          notebook.title,
                          style: TextStyle(
                            fontSize: 18,
                            fontWeight: notebook.isBold ? FontWeight.bold : FontWeight.w500,
                            fontStyle: notebook.isItalic ? FontStyle.italic : FontStyle.normal,
                            color: notebook.color,
                          ),
                          textAlign: TextAlign.center,
                          maxLines: 2,
                          overflow: TextOverflow.ellipsis,
                        ),
                      ),
                      const SizedBox(height: 8),
                      Text(
                        '${notebook.subdivisions.length} ${notebook.subdivisions.length == 1 ? 'subdivisão' : 'subdivisões'}',
                        style: TextStyle(fontSize: 12, color: Colors.grey[600]),
                      ),
                    ],
                  ),
                ),
              );
            },
          );
        },
      ),
      floatingActionButton: FloatingActionButton.extended(
        onPressed: () {
          final provider = Provider.of<NotebookProvider>(context, listen: false);
          final notebook = Notebook(
            title: 'Novo Caderno',
            color: Colors.blue,
            emoji: '📘',
          );
          provider.addNotebook(notebook);
        },
        icon: const Icon(Icons.add),
        label: const Text('Novo Caderno'),
      ),
    );
  }
}
